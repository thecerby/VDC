INSERT INTO AbpNotificationSubscriptions (Id, CreationTime, CreatorUserId, EntityId, EntityTypeAssemblyQualifiedName, EntityTypeName, NotificationName, TenantId, UserId)
    SELECT newid(), getdate(), NULL, NULL, NULL, NULL, 'Vehicles.StatusChanged', u.TenantId, u.Id
    FROM AbpUsers u
        LEFT OUTER JOIN AbpNotificationSubscriptions n ON u.Id = n.UserId AND n.NotificationName = 'Vehicles.StatusChanged'
    WHERE u.TenantId IS NOT NULL
    AND n.Id IS NULL
